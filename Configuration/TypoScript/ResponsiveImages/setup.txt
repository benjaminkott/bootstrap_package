page {
    includeJSFooterlibs {
        responsiveimages = EXT:bootstrap_package/Resources/Public/JavaScript/Dist/jquery.responsiveimages.min.js
        responsiveimages.if {
            value = 1
            equals = {$plugin.bootstrap_package.settings.images.tag}
        }
        blazy = EXT:bootstrap_package/Resources/Public/JavaScript/Libs/blazy.min.js
        blazy.if {
            value = 1
            equals = {$plugin.bootstrap_package.settings.images.tag}
            negate = 1
        }
    }

    jsFooterInline {
        # b-lazy
        50 = COA
        50 {
            if {
                value = 1
                equals = {$plugin.bootstrap_package.settings.images.tag}
                negate = 1
            }
            10 = TEXT
            10.value (
                ;(function() {
                    var bLazy = new Blazy({
                     selector:'.lazyload',
                     successClass:'loaded',
                     success:function(){
                        $(window).trigger('loaded.bk2k.responsiveimage');
                     },
                     error:function(){
                     },
                     loadInvisible: true
                    // container:'body'
                    });
                    $(window).on('beforeprint.bk2k.responsiveimage', function(){
                        bLazy.load('.lazyload', true);
                        $(window).trigger("readytoprint.bk2k.responsiveimage");
                    });
                })();
            )
        }
    }
